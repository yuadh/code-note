## 按钮控制LED

初始化

```c
void Init_Port()
{
  P1SEL &= ~0x1b;     //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;      //P1_0、P1_1、P1_3和P1_4端口输出
  
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口 //0000 0100 
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  P0SEL &= ~0x02;     //P0_1作为通用I/O端口
  P0DIR &= ~0x02;     //P0_1端口输入
  P0INP &= ~0x02;     //P0_1设置为上拉/下拉模式
  P2INP &= ~0x20;     //P0_1设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}
```

按键控制

```js
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      D3 = ~D3;           //切换D3灯的开关状态
    }
  }    
  
  if(SW2 == 0)   
  {
    Delay(100);
    if(SW2 == 0)
    {
      while(SW2 == 0);
      D4 = ~D4;
    }
  }    
}
```

升级版按键控制轮询

```js
/*=======================键值处理函数=========================*/
void DoWork_Keys()
{
  switch(K_Num)
  {
    case 1:         //键值“1”处理
      D3 = 1;
    break;
    case 2:         //键值“2”处理
      D4 = 1;
    break;
    case 3:         //键值“3”处理
      D5 = 1;
    break;
    case 4:         //键值“4”处理
      D6 = 1;
    break;
    case 0:         //键值“0”处理
      D3 = 0;
      D4 = 0;
      D5 = 0;
      D6 = 0;
    break;
  }
}

/*=======================按键扫描函数=========================*/
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      K_Num++;            //改变键值状态
      if(K_Num == 5)
      {
        K_Num = 0;
      }
      DoWork_Keys();      //执行按键操作
    }
  }     
}

```

家控灯

```c
/*=======================键值处理函数=========================*/
void DoWork_SW1()
{
  switch(K_Num)
  {
    case 1:         //状态1
      D6 = 1;
      D5 = 0;
    break;
    case 2:         //状态2
      D6 = 0;
      D5 = 1;
    break;
    case 0:         //状态3
      D6 = 0;
      D5 = 0;
      D4 = 0;
      D3 = 0;
    break;
      
  }
}

/*=======================按键扫描函数=========================*/
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      K_Num++;            //键值变换
      if(K_Num == 3)
      {
        K_Num = 0;
      }
      DoWork_SW1();       //键值处理
    }
  }    
  
  if(SW2 == 0)   
  {
    Delay(100);
    if(SW2 == 0)
    {
      while(SW2 == 0);
      if(K_Num == 1)        //状态1
      {
        D4 = ~D4;
      }
      else if(K_Num == 2)   //状态2
      {
        D3 = ~D3;
      }
    }
  }    
}

```
