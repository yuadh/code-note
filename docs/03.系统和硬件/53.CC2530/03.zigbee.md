```c
#include "ioCC2530.h"
#include "gpio.h"
#include "uart0.h"

unsigned char i;
char SendPacket[]={0x13,0x61,0x88,0x00,0x00,0x22,0xEF,0xBE,0x20,0x50,
                   'h','e','l','l','o',' ','\r','\n'};
void init_RF()
{
  FRMCTRL0 |= (0x20 | 0x40);//硬件CRC以及AUTO_ACK使能
  TXFILTCFG = 0x09;//设置TX抗混叠过滤器以获得合适的带宽
  AGCCTRL1 = 0x15;//调整AGC目标值
  FSCAL1 = 0x00;//获得最佳的EVM
  RFIRQM0 |= (1<<6);// RXPKTDONE 中断位使能
  IEN2 |= (1<<0);//  RF 中断使能
  EA = 1;//开中断
  FREQCTRL = 0x0b;//信道选择，选择11信道 
  PAN_ID0 = 0x22; 
  PAN_ID1 = 0x00;//目标地址过滤期间使用的PANID  
  RFST = 0xed; //清除RXFIFO缓冲区并复位解调器
  RFST = 0xe3; //为RX使能并校准频率合成器
  FRMFILT0 &= ~(1<<0);//禁止帧过滤
}
void RF_Send(char *pstr,char len)
{
  RFST = 0xe3; //为RX使能并校准频率合成器 
  // TX_ACTIVE | SFD 
  while (FSMSTAT1 & ((1<<1) | (1<<5))); 
  RFIRQM0 &= ~(1<<6);//禁止RXPKTDONE中断 
  IEN2 &= ~(1<<0);//禁止RF中断 
  RFST = 0xee; // 清除TXFIFO缓存
  RFIRQF1 = ~(1<<1);// 清除 TXDONE 中断
  for(i=0;i<len;i++)
    {
       RFD=pstr[i];
    }  
  RFIRQM0 |= (1<<6); // 0100 0000打开RX中断
  IEN2 |= (1<<0);//打开RF中断
  RFST = 0xe9;//校准后使能TX 
  while (!(RFIRQF1 & (1<<1)));//等待传输结束 
  RFIRQF1 = ~(1<<1);//清除 TXDONE状态 
  D5=~D5;//D5灯状态改变
  Delay(200);
  Delay(200);
}

void main(void)
{ 
  EA = 0;//关闭总中断
  init_port();//初始化普通I/O口，即灯和按键
  Clock_32M();//设置时钟频率为32M
  init_RF();//初始化RF
  SHORT_ADDR0 = 0x50;
  SHORT_ADDR1 = 0x20;//目标地址过滤期间使用的短地址
  EA = 1; //中断使能
  while(1) 
  {
    RF_Send(SendPacket,19); //RF发送数据包
    Delay(2000);			
  }
}
```

```c
#pragma vector=RF_VECTOR
__interrupt void Service_RF() 
{
  unsigned char  i; 
  IEN2 &= ~0x01; //关中断
  if (RFIRQF0 & (1<<6)) //0100 0000 接收帧结束
  {
    len = RFD;//接收帧长度		
    while (len>0) 
    {//只要后面还有数据那么就把他都从接受缓冲区取出来
      ch=RFD;
      if(len<11&&len>2)
      {
        buf[i] = ch;
        i++;
      }
      len--;
    }
    S1CON = 0; // 清RF中断  
    RFIRQF0 &= ~(1<<6); //清 RXPKTDONE中断
    D5 = ~D5;   //D5等状态改变
  }
  IEN2 |= (1<<0);
  uart0_SendStr(buf); //串口发送接收到的无线数据到串口助手显示
  i=0;
}

void main(void)
{ 
  init_port();
  init_UART0();
  EA = 0;//关闭总中断
  Clock_32M();//设置时钟频率为32M
  init_RF();//初始化RF

  SHORT_ADDR1 = 0xEF;//目标地址过滤期间使用的短地址
```
