## 官方示例



## 01-LED灯

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1

/*==================延时函数=====================*/
void Delay(unsigned int t)
{
  while(t--);
}

/*==================主函数======================*/
void main()
{
  P1SEL &= ~0x03;
  P1DIR |= 0x03;
  D3 = 0;
  D4 = 0;
  
  while(1)
  {
    D3 = 1;           //D3灯点亮
    Delay(60000);     //延时
    D4 = 1;           //D4灯点亮
    Delay(60000);     //延时
    D3 = 0;           //D3灯熄灭
    Delay(60000);     //延时
    D4 = 0;           //D4灯熄灭
    Delay(60000);     //延时
  }
}
```

```c
  D6 = 0;

  unsigned char i;

  D5 = 1; 
  Delay(60000);
  D4 = 1; 
  Delay(60000);
  D3 = 1;
  Delay(60000);
  D6 = 0;
  Delay(60000);
  D5 = 0; 
  Delay(60000);
  D4 = 0; 
  Delay(60000);
  D3 = 0;
         
  D6 = 1;
  Delay(60000);
  D5 = 1; 
  Delay(60000);
  D4 = 1; 
  Delay(60000);
  D3 = 1;
  Delay(60000);
  D3 = 0;
  Delay(60000);
  D4 = 0; 
  Delay(60000);
  D5 = 0;
```

## 03-按键控制LED

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4
#define SW1 P1_2
#define SW2 P0_1

/*=======================简单的延时函数========================*/
void Delay(unsigned int t)
{
  while(t--);
}
/*======================端口初始化函数========================*/
void Init_Port()
{
  P1SEL &= ~0x1b;     //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;      //P1_0、P1_1、P1_3和P1_4端口输出
  
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  P0SEL &= ~0x02;     //P0_1作为通用I/O端口
  P0DIR &= ~0x02;     //P0_1端口输入
  P0INP &= ~0x02;     //P0_1设置为上拉/下拉模式
  P2INP &= ~0x20;     //P0_1设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}

/*=======================按键扫描函数=========================*/
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      D3 = ~D3;           //切换D3灯的开关状态
    }
  }    
  
  if(SW2 == 0)   
  {
    Delay(100);
    if(SW2 == 0)
    {
      while(SW2 == 0);
      D4 = ~D4;
    }
  }    
}

/*=========================主函数============================*/
void main()
{
  Init_Port();         //端口初始化
  while(1)
  {
    Scan_Keys();        //反复扫描按键状态
  }
}
```

## 04-按键循环LED

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4
#define SW1 P1_2

unsigned char K_Num = 0;      //定义键值变量

/*=======================简单的延时函数========================*/
void Delay(unsigned int t)
{
  while(t--);
}
/*======================端口初始化函数========================*/
void Init_Port()
{
  P1SEL &= ~0x1b;     //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;      //P1_0、P1_1、P1_3和P1_4端口输出
  
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}

/*=======================键值处理函数=========================*/
void DoWork_Keys()
{
  switch(K_Num)
  {
    case 1:         //键值“1”处理
      D3 = 1;
    break;
    case 2:         //键值“2”处理
      D4 = 1;
    break;
    case 3:         //键值“3”处理
      D5 = 1;
    break;
    case 4:         //键值“4”处理
      D6 = 1;
    break;
    case 0:         //键值“0”处理
      D3 = 0;
      D4 = 0;
      D5 = 0;
      D6 = 0;
    break;
  }
}

/*=======================按键扫描函数=========================*/
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      K_Num++;            //改变键值状态
      if(K_Num == 5)
      {
        K_Num = 0;
      }
      DoWork_Keys();      //执行按键操作
    }
  }     
}

/*=========================主函数============================*/
void main()
{
  Init_Port();         //端口初始化
  while(1)
  {
    Scan_Keys();        //反复扫描按键状态
  }
}
```

## 06-按键中断

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4

/*=======================简单的延时函数========================*/
void Delay(unsigned int t)
{
  while(t--);
}
/*======================端口初始化函数========================*/
void Init_Port()
{
  P1SEL &= ~0xb;    //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;    //P1_0、P1_1、P1_3和P1_4端口输出
 
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}

/*========================灯光闪烁函数=========================*/
void LED_Running()
{
  D3 = ~D3;
  Delay(60000);
}

/*====================外部中断初始化函数======================*/
void Init_INTP1()
{
  IEN2 |= 0x10;     //使能P1端口组外部中断（IEN2不能位操作）
  P1IEN |= 0x04;    //使能P1_2引脚外部中断
  PICTL &= ~0x02;   //配置P1_0到P1_3上升沿触发
  EA = 1;           //打开总中断
}

/*=====================外部中断服务函数=======================*/
#pragma vector = P1INT_VECTOR
__interrupt void INTP1_Service()
{
  if((P1IFG & 0x04) == 0x04)    //判断是否P1_2引脚产生中断
  {
    D6 = ~D6;
    P1IFG &= ~0x04;             //清除P1_2引脚中断标志
  }
  P1IF = 0;                     //清除P2端口组中断标志
}

/*==========================主函数============================*/
void main()
{
  Init_Port();        //端口初始化
  Init_INTP1();       //初始化外部中断
  while(1)
  {
    LED_Running();    //运行灯光闪烁主程序
  }
}
```

## 0308class

- EA - 总中断
- PXIE - 组中断
- PXIEN - 组中各中断
- PICTL - 触发方式选择

------------------------

中断标志位

- PXIF - 总中断标志
- PXIFG - 组中各组中断标志

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4
//0.设置使用设备 PXSEL , 1.外设 0.通用IO
//1.设置输入输出设备 1为输出 0为输入  2.设置上拉下拉模式 1为设置三态(max) 0为上拉下拉
//  [3].设置上拉或下拉 0为上拉输入 1为下拉输入
unsigned char sta=0;
void Init_port(){
  // 0001 1011  初始化输出端口
  P1SEL &= ~0x1b;    //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;    //P1_0、P1_1、P1_3和P1_4端口输出
  // 0000 0100
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}
void delay(int i){
  //延时操作
  int j;
  while(i--){
    for(j=0;j<255;j++);
  }
}
void LedSta(){
  //灯亮的状态
  switch(sta){
    case 1:
      D5=1;D6=0;D3=0;D4=0;
      break;
    case 2:
      D5=0;D6=1;D3=0;D4=0;
      break;
    case 3:
      D5=0;D6=0;D3=1;D4=0;
      break;
    case 4:
      D5=0;D6=0;D3=0;D4=1;
      break;
    default:
      D5=0;D6=0;D3=0;D4=0;
  }
}
void Init_IntP1(){
  //初始化中断
  //0001 0000
  IEN2 |= 0x10;     //使能P1端口组外部中断（IEN2不能位操作）
  P1IEN |= 0x04;    //使能P1_2引脚外部中断
  PICTL &= ~0x02;   //配置P1_0到P1_3上升沿触发
  EA = 1;           //打开总中断
}
//端口1 中断处理程序
#pragma vector = P1INT_VECTOR
__interrupt void INTP1_Service(){
  if((P1IFG & 0x04) == 0x04){
      if(sta<4){
        sta++;
      }else{
        sta=0;
      }
      P1IFG &= ~0x04; //清除中断端口标志位
  }
  P1IF = 0; //清楚中断组标志位
}
void main(){
  //初始化需要用到的端口
  void Init_port();
  void Init_IntP1();
  while(1){
    LedSta();
  }
}
```
