## 官方示例



## 01-LED灯

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1

/*==================延时函数=====================*/
void Delay(unsigned int t)
{
  while(t--);
}

/*==================主函数======================*/
void main()
{
  P1SEL &= ~0x03;
  P1DIR |= 0x03;
  D3 = 0;
  D4 = 0;
  
  while(1)
  {
    D3 = 1;           //D3灯点亮
    Delay(60000);     //延时
    D4 = 1;           //D4灯点亮
    Delay(60000);     //延时
    D3 = 0;           //D3灯熄灭
    Delay(60000);     //延时
    D4 = 0;           //D4灯熄灭
    Delay(60000);     //延时
  }
}
```

## 03-按键控制LED

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4
#define SW1 P1_2
#define SW2 P0_1

/*=======================简单的延时函数========================*/
void Delay(unsigned int t)
{
  while(t--);
}
/*======================端口初始化函数========================*/
void Init_Port()
{
  P1SEL &= ~0x1b;     //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;      //P1_0、P1_1、P1_3和P1_4端口输出
  
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  P0SEL &= ~0x02;     //P0_1作为通用I/O端口
  P0DIR &= ~0x02;     //P0_1端口输入
  P0INP &= ~0x02;     //P0_1设置为上拉/下拉模式
  P2INP &= ~0x20;     //P0_1设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}

/*=======================按键扫描函数=========================*/
void Scan_Keys()
{
  if(SW1 == 0)            //发现有SW1按键信号
  {
    Delay(100);           //延时片刻，去抖动处理
    if(SW1 == 0)          //确认为SW1按键信号
    {
      while(SW1 == 0);    //等待按键松开
      D3 = ~D3;           //切换D3灯的开关状态
    }
  }    
  
  if(SW2 == 0)   
  {
    Delay(100);
    if(SW2 == 0)
    {
      while(SW2 == 0);
      D4 = ~D4;
    }
  }    
}

/*=========================主函数============================*/
void main()
{
  Init_Port();         //端口初始化
  while(1)
  {
    Scan_Keys();        //反复扫描按键状态
  }
}
```

## 06-按键中断

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4

/*=======================简单的延时函数========================*/
void Delay(unsigned int t)
{
  while(t--);
}
/*======================端口初始化函数========================*/
void Init_Port()
{
  P1SEL &= ~0xb;    //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;    //P1_0、P1_1、P1_3和P1_4端口输出
 
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式
  P2INP &= ~0x40;     //P1_2设置为上拉
  
  D3 = 0;          
  D4 = 0;          
  D5 = 0;           
  D6 = 0; 
}

/*========================灯光闪烁函数=========================*/
void LED_Running()
{
  D3 = ~D3;
  Delay(60000);
}

/*====================外部中断初始化函数======================*/
void Init_INTP1()
{
  IEN2 |= 0x10;     //使能P1端口组外部中断（IEN2不能位操作）
  P1IEN |= 0x04;    //使能P1_2引脚外部中断
  PICTL &= ~0x02;   //配置P1_0到P1_3上升沿触发
  EA = 1;           //打开总中断
}

/*=====================外部中断服务函数=======================*/
#pragma vector = P1INT_VECTOR
__interrupt void INTP1_Service()
{
  if((P1IFG & 0x04) == 0x04)    //判断是否P1_2引脚产生中断
  {
    D6 = ~D6;
    P1IFG &= ~0x04;             //清除P1_2引脚中断标志
  }
  P1IF = 0;                     //清除P2端口组中断标志
}

/*==========================主函数============================*/
void main()
{
  Init_Port();        //端口初始化
  Init_INTP1();       //初始化外部中断
  while(1)
  {
    LED_Running();    //运行灯光闪烁主程序
  }
}
```

## 07-定时器

```c
#include "ioCC2530.h"

#define D3  P1_0
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4

unsigned char count = 0;      //长定时累计变量

/*======================端口初始化函数========================*/
void Init_Port()
{
  //初始化LED灯的I/O端口
  P1SEL &= ~0x1b;   //P1_0、P1_1、P1_3和P1_4作为通用I/O端口
  P1DIR |= 0x1b;    //P1_0、P1_1、P1_3和P1_4端口输出
  //关闭所有的LED灯
  P1 &= ~0x1b;
}

/*=====================定时器1初始化函数======================*/
void Init_Timer1()
{
  T1CC0L = 0xd4;    //16MHz时钟128分频定时100ms 13 
  T1CC0H = 0x30;    //设先填低8位，再填高8位
  T1CCTL0 |= 0x04;  //开启通道0的输出比较模式
  T1IE = 1;         //使能定时器1中断	时钟使能
  T1OVFIM = 1;      //使能定时器1溢出中断
  EA = 1;           //使能总中断
  T1CTL = 0x0e;     //分频系数是128,模模式
}

/*====================定时器1中断服务函数=====================*/
#pragma vector = T1_VECTOR
__interrupt void Timer1_Sevice()
{
  T1STAT &= ~0x01;      //清除定时器1通道0中断标志
  count++;              //累加定时周期
  if(count%5 == 0)      //定时0.5秒时间到
  {
    D3 = ~D3;
  }
  if(count%10 == 0)     //定时1秒时间到
  {
    D4 = ~D4;
  }
  if(count == 50)       //定时5秒时间到
  {
    D5 = ~D5;
    count = 0;
  }
}

/*==========================主函数============================*/
void main()
{
  Init_Port();        //端口初始化
  Init_Timer1();       //初始化定时器1
  while(1);
}

```



## 08-长按控制LED

```c
#define D4  P1_1
#define D5  P1_3
#define D6  P1_4
#define SW1 P1_2

void Delay(unsigned int t)

  //关闭所有的LED灯
  P1 &= ~0x1b;
  //初始化按键
  P1SEL &= ~0x04;     //P1_2作为通用I/O端口
  P1DIR &= ~0x04;     //P1_2端口输入
  P1INP &= ~0x04;     //P1_2设置为上拉/下拉模式

  T1CC0H = 0x30;    //设先填低8位，再填高8位
  T1CCTL0 |= 0x04;  //开启通道0的输出比较模式
  T1IE = 1;         //使能定时器1中断
  T1OVFIM = 1;      //使能定时器1溢出中断
  EA = 1;           //使能总中断
  T1CTL = 0x0e;     //分频系数是128,模模式
}

/*====================定时器1中断服务函数=====================*/
#pragma vector = T1_VECTOR
__interrupt void Timer1_Sevice()
{
  T1STAT &= ~0x01;      //清除定时器1通道0中断标志
  if(K_Press != 0)      //按键按下
  {
    count_t++;          //计算按下按下的时间值
  }
}

/*====================按键扫描处理函数========================*/

void Scan_Keys()

{
  if(SW1 == 0)
  {
    Delay(100);         //去抖动处理
    if(SW1 == 0)
    {
      K_Press = 1;      //标志按键正在按下
      while(SW1 == 0);  //等待按键松开
      K_Press = 0;      //标志按键已经松开
      
      if(count_t > 10)  //按键长按
      {
        D6 = ~D6;
      }
      else              //按键短按
      {
        D4 = ~D4;
      }
      count_t = 0;      //按键计数值清零
    }
  }
}

/*==========================主函数============================*/
void main()
{
  Init_Port();          //端口初始化
  Init_Timer1();        //初始化定时器1
  while(1)
  {
    Scan_Keys();        //扫描按键
  }
```
