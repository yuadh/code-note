## 程序代码

### 接收处理函数

```c
/**
  desc:接收到AF数据的处理函数
*/
static void GenericApp_MessageMSGCB(afIncomingMSGPacket_t *MSGpkt)
{
  static uint16 p2pCnt = 0;
  static uint16 bcCnt = 0;
  static uint16 gcCnt = 0;

  switch( MSGpkt->clusterId )  // 判断接收到的数据包的 Cluster ID 类型
  {
    case CLUSTER_P2P:
      p2pCnt++;  // 接收到 P2P 数据包，进行计数
                 // 把接收到的数据显示在屏幕上，并翻转 LED
      HalUARTWrite(0,(unsigned char *)MSGpkt->cmd.Data,p2pCnt);
      HalLedSet(HAL_LED_1, HAL_LED_MODE_TOGGLE);
      break;

    case CLUSTER_BROADCAST:
      bcCnt++;  // 接收到广播数据包，进行计数
      HalUARTWrite(0, (unsigned char *)MSGpkt->cmd.Data,bcCnt);
      HalLedSet(HAL_LED_2, HAL_LED_MODE_TOGGLE);
      // 将得到的数据显示到串口助手上
      break;

    case CLUSTER_GROUPCAST:
      gcCnt++;  // 接收到组播数据包，进行计数
      HalUARTWrite(0,(unsigned char *)MSGpkt->cmd.Data,gcCnt);
      HalLedSet(HAL_LED_3, HAL_LED_MODE_TOGGLE);
      break;
    default:
      break;
  }
}
```

### 三种通信发送方式

```c
/**
  desc:p2p通信方式
  uint16 destNwkAddr,//目标设备的网络地址
  uint16 cid,//Cluster ID，后续详细讲解
  uint8 len,//数据长度和数据内容
  uint8 *data
*/
static void GenericApp_AF_P2P(uint16 destNwkAddr,uint16 cid,uint8 len,uint8 *data){
  afAddrType_t dstAddr; //寻址信息配置
  static uint8 transferId = 0;//传输 ID，是数据包的标识符，暂时无须理
  /* Destination */
  dstAddr.addrMode = afAddr16Bit;//设置目标地址模式为 16 位网络地址，表示使用 P2P 的通信方式
  dstAddr.addr.shortAddr = destNwkAddr;  //目标设备网络地址
  dstAddr.endPoint = GENERICAPP_ENDPOINT;  //目标设备的应用端点号
  transferId++;
  AF_DataRequest( &dstAddr,
                  &GenericApp_epDesc,//之前创建的简单描述符
                  cid,
                  len,
                  data,
                  &transferId,
                  AF_DISCV_ROUTE,//进行路由扫描操作用于建立发送数据报文的通信路径，暂时无须理解
                  AF_DEFAULT_RADIUS);//指路由跳转级数，暂时无须理解
}
/**
  desc:广播通信

*/
static void GenericApp_AF_Broadcast(uint16 cid,uint8 len, uint8 *data){
  afAddrType_t dstAddr;
  static uint8 transferId = 0;
  /* Destination */
  dstAddr.addrMode = afAddrBroadcast;   // 使用广播模式
  dstAddr.addr.shortAddr = 0xFFFF; // 广播地址
  dstAddr.endPoint = GENERICAPP_ENDPOINT; // 目标设备的端点号
  /* Transfer id */
  transferId++;
  /* Send */
  AF_DataRequest(&dstAddr,
                 &GenericApp_epDesc,
                 cid,
                 len, data,
                 &transferId,
                 AF_TX_OPTIONS_NONE,
                 AF_DEFAULT_RADIUS );
}

/**
  desc：组播通信
  uint16 groupId,//组 ID
  uint16 cid,//ClusterID
  uint8 len, uint8 *data//数据的长度和内容
*/

static void zclSampleSw_AF_Groupcast(uint16 groupId,uint16 cid,uint8 len, uint8 *data)
{
  afAddrType_t dstAddr;
  static uint8 transferId = 0;
  /* Destination */
  dstAddr.addrMode = afAddrGroup;// 使用组播通信模式
  dstAddr.addr.shortAddr = groupId;// 组 ID
  dstAddr.endPoint = GENERICAPP_ENDPOINT;  // 组中的设备的端点号
  /* Transfer id */
  transferId++;
  /* Send */
  AF_DataRequest(&dstAddr,
                 &GenericApp_epDesc,
                 cid,
                 len,
                 data,
                 &transferId,
                 AF_TX_OPTIONS_NONE,
                 AF_DEFAULT_RADIUS );
}
```

协调器按下按键SW2后，广播发送数据“下周考试”

**事件：有接收到数据包事件**

```c
 case AF_INCOMING_MSG_CMD:
          GenericApp_MessageMSGCB((afIncomingMSGPacket_t *)MSGpkt);

```

**按键触发，广播数据**

```c
if ( keys & HAL_KEY_SW_6 )//SW2按钮//广播按有效
    {
      // 亮灯 D6 ，提示广播发送
      HalLedBlink(HAL_LED_2,1,HAL_LED_DEFAULT_DUTY_CYCLE, HAL_LED_DEFAULT_FLASH_TIME);
      GenericApp_AF_Broadcast(CLUSTER_BROADCAST,14,"下周考试\r\n");//广播发送
    }
```
